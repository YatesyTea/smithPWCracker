49 int GetIndexOfWorkingCandidateInThisList(candidateList thislist)
50 {
51  extern candidateSolution workingCandidate;
52  char wcvalues[N];
53  int place=0, which=0, mismatches=N;
54 
55  //copy peg values from working candidate into conveniently named array becuase it is faster this way
56  for(place=0;place<N;place++)
57  wcvalues[place] = workingCandidate.variableValues[place];
58 
59  //start at the end of the list
60  which = thislist.size -1;
61  //and loop through until we have found a match or reached the beginning
62  while (which >=0)
63  {
64  // start looking peg by peg, stopping as soon as there is a mismatch
65  for (place=0,mismatches = 0; (place < N)&& (mismatches ==0);place++)
66  if (thislist.listEntries[which].variableValues[place] != wcvalues[place])
67  mismatches++;
68  if (mismatches>0) //no mismatches found so move to next one in the list
69  which --;
70  else
71  break;
72  }
73  //end of which loop, now just need to look and see if we found it
74  if (which < 0)
75  which = NOTFOUND;
76  return which;
77 }

262 void RemoveFromListParam1_CandidateSolutionAtIndexParam2( candidateList *listToEdit, int solutionIndex)
263 {
264 
265  int numberOfEntriesInThisList,lastEntry; //declare int to hold number of solutions currently in the list
266  int copyingfrom, copyingto;// and two more to use when we shuffle things along
267  int row=0;
268 
269  //next we'll start off with a series of error checks before we do anything
270  //each of these is serous enough to cause us to quit the programme
271 
272  //first of all, have we been given the address of a valid candidate list
273  if( listToEdit == (candidateList *)(NULL))
274  PrintThisMessageAndExit("first parameter passed was not a pointer to a valid list");
275 
276  // then of all check that the index we are given is valid
277  else if ( (solutionIndex >= SIZEOFSOLUTIONLIST) || (solutionIndex <0))
278  PrintThisMessageAndExit("oops - tried to remove a solution that lies beyond the bounds of our list");
279 
280  // if we've got here we know it is safe to try and access the lists
281  else
282  {
283  // last couple of tests
284  //find out how many entries are in the list
285  numberOfEntriesInThisList = listToEdit->size;
286  lastEntry = numberOfEntriesInThisList -1;
287 
288  // check that the list is not already empty
289  if (numberOfEntriesInThisList<1)
290  PrintThisMessageAndExit("oops - tried to remove an entry from an empty list");
291  //check the index isn't too high
292  if ( solutionIndex > lastEntry)
293  PrintThisMessageAndExit("oops - tried to remove a solution that does not exist in our list");
294 
295 
296 
297  // ok, then we can proceed
298  //printf("removing solution at index %d \n", solutionIndex);
299  //if the one to remove is the last on the list, its all very simple
300  //otherwisde we have to shuffle all the rest along and then delete what is now a duplicate lasdt entry
301  if(solutionIndex < lastEntry)
302  {
303  //start with the one to be deleted, and overwrite the contents with the solution to the right
304  //then move right and repeat
305  for(copyingto = solutionIndex; copyingto < lastEntry;copyingto++)
306  {
307  copyingfrom = copyingto + 1;//overwrite with the contents of the solution to the right in the list
308  listToEdit->listEntries[copyingto].size = listToEdit->listEntries[copyingfrom].size;
309  listToEdit->listEntries[copyingto].score = listToEdit->listEntries[copyingfrom].score;
310  for (row=0;row < N;row++)
311  listToEdit->listEntries[copyingto].variableValues[row] = listToEdit->listEntries[copyingfrom].variableValues[row];
312 
313  }//end of copying loop
314  }//end if code that deals with shuffling situation
315 
316  //now delete the last entry - code should skip to here if that's the one we're asked to remove
317  numberOfEntriesInThisList --;
318 
319  listToEdit->listEntries[numberOfEntriesInThisList ].size = 0;
320  listToEdit->listEntries[numberOfEntriesInThisList ].score = 0;
321  for ( row=0;row < N;row++)
322  listToEdit->listEntries[numberOfEntriesInThisList].variableValues[row] = UNUSED;
323  //and reduce the count of number of things in the list
324  listToEdit->size -- ;
325 
326 
327  }//end of code for valid call
328 }//end of generic removefunction
